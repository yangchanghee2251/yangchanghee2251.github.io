def solution(prices):
    answer = []
    length=len(prices)
    num=0
    while prices:
        num+=1

주식의 상승장일 때만 쌓는 알고리즘을 생각해 풀었을 때 코드이다.  
효율성은 0% 정확도는 100%이다.

```
from collections import deque

def solution(prices):
    dq1=[]
    dq2=deque()
    length=len(prices)
    answer=[0 for k in range(length)]
    for n in range(length):
        if length==n+1:
            continue
        if prices[n] <= prices[n+1]:
            dq1.append(prices[n]) # stock
            dq2.append(n+1) # startpoints
        else:
            answer[n]=1
            if dq1:
                while prices[n+1]<dq1[-1]:
                    dq1.pop()
                    sp=dq2.pop()
                    answer[sp-1]=n+2-sp
                    if not dq1:
                        break
    #print(dq1,dq2)
    while dq1:
        dq1.pop()
        sp=dq2.popleft()
        answer[sp-1]=length-sp
    return answer
```
deque를 사용해 최대한 효율적으로 만들어 풀었다. 올클!
